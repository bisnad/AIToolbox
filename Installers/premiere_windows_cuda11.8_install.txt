# Manual Installation Instructions for Premiere Conda Environment
# Windows GPU (CUDA 11.8) Version

# Prerequisites
# Install Anaconda (https://www.anaconda.com/download/success)
# Install Cuda 11.8 (from local installer or https://developer.nvidia.com/cuda-11-8-0-download-archive?target_os=Windows&target_arch=x86_64)
# Install Cudnn 8.7.0.84 (from local archive or https://developer.nvidia.com/downloads/c118-cudnn-windows-8664-87084cuda11-archivezip)
# Clone or Download the AIToolbox Github Repository (https://github.com/bisnad/AIToolbox)

# create conda environment
conda create --name premiere python=3.10

# activate conda environment
conda activate premiere

# add conda forge as additional default channel 
conda config --add channels conda-forge

# in the anaconda command, make sure to change into the directory 
# that contains the fbx-2020.3.7-cp310-none-win_amd64.whl file

# install conda packages
conda install mamba
mamba install pytorch==2.2.2 torchvision==0.17.2 torchaudio==2.2.2 pytorch-cuda=11.8 -c pytorch -c nvidia -y
mamba install spyder -y
mamba install numpy==1.26.4 -y
mamba install transformers -y
mamba install diffusers -y
mamba install datasets -y
mamba install sentencepiece -y
mamba install evaluate -y
mamba install matplotlib -y
mamba install scikit-learn -y
mamba install pandas -y
mamba install h5py -y
mamba install librosa -y
mamba install python-sounddevice -y
mamba install pysoundfile -y
mamba install tqdm -y
mamba install pyqt -y
mamba install pyqtgraph -y
mamba install pyopengl -y
mamba install transforms3d -y
mamba install pybullet -y
mamba install gym -y
mamba install ffmpeg-python -y
mamba install opencv -y
mamba install vispy -y
mamba install jupyterlab -y

# install pip packages
pip install einops-exts
pip install python-osc
pip install simpleaudio
pip install vocos
pip install fbx-2020.3.7-cp310-none-win_amd64.whl
